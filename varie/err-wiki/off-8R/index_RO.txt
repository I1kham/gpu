<script language="javascript">
var OFF_isBlocking = "N";

function onRheaBootstrapFinished()
{
	notifyGPUThatImHere();
	menuLang_setup (lang_listOfAvailLang);

	if (OFF_isBlocking == "Y")
	{
		rheaHideElem (rheaGetElemByID("btnClose"));
		rhea.onEvent_cpuStatus = function (statusID, statusStr, flag16)
		{
			if (statusID == 5) //CPU e' andata in "ini-check, probabilmente si e' autoresettata
				onBtnClose();	
		}
	}
	else
	{
		autoclose_on_timer(60);
	}
}

function notifyGPUThatImHere()
{
	//notifico GPU che sono entrato in questa pagina
	rhea.ajax ("GUINotify", {"p0":1002}).then( function(result)
	{
	})
	.catch( function(result)
	{
		setTimeout ( function() { notifyGPUThatImHere(); }, 1000);
	});
}

var already_closing = 0;
function autoclose_on_timer (timeLeft_sec)
{
	if (already_closing)
		return;
	if (timeLeft_sec <= 0)
		onBtnClose();
	else
	{
		rheaSetElemHTML(rheaGetElemByID("btnCloseText"), "ÎNCHIDE (" +timeLeft_sec +")");
		setTimeout ( function() { autoclose_on_timer(timeLeft_sec-1); }, 1000);
	}
}

function onBtnClose()
{
	already_closing = 1;
	rheaHideElem (rheaGetElemByID("btnClose"));
	
	//notifico GPU che l'utente vuole chiudere la pagina
	rhea.ajax ("GUINotify", {"p0":1001}).then( function(result)
	{
	})
	.catch( function(result)
	{
		setTimeout ( function() { onBtnClose(); }, 1000);
	});
}


</script>
